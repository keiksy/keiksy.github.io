def letterCombinations(self, digits: str) -> List[str]:
		words = ['abc','def','ghi','jkl','mno','pqrs','tuv','wxyz']
    ans = []
    if len(digits)==0:
      	return ans
        
    def helper(position, temp):
      	if position==len(digits):
          	ans.append(temp)
            return
    		word = words[int(digits[position])-2]
        for c in word:
          	temp+=c
            helper(position+1, temp)
            temp = temp[:-1]        
    helper(0,'')
    return ans
